Get Started in 2 Minutes

Terminal 
# Install via NPM (Recommended)
npm install -g vacuum-sol
vacuum --version
vacuum --help

# Or clone from source
git clone https://github.com/Don-Vicks/vaccum.git
cd vaccum && npm install && npm run build
npm link

# Configure
export SOLANA_RPC_URL=https://api.devnet.solana.com
export TREASURY_ADDRESS=
export OPERATOR_KEYPAIR_PATH=~/.vacuum/operator.json

# Run!
vacuum scan
vacuum check --all
vacuum reclaim --dry-run
vacuum report
